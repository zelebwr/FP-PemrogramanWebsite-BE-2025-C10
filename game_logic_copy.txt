    <script>
        // ============ CONFIGURATION ============
        const API_BASE = 'http://localhost:4000';

        // ============ GAME STATE ============
        let gameData = null;
        let teams = [];
        let currentTeam = 0;
        let currentClue = null;
        let usedClues = new Set();
        let totalClues = 0;
        let selectedGameId = null;

        // ============ INITIALIZATION ============
        document.addEventListener('DOMContentLoaded', () => {
            loadGames();
            setupListeners();
        });

        function setupListeners() {
            // Team count change
            document.getElementById('teamCount').addEventListener('change', (e) => {
                const count = parseInt(e.target.value);
                const container = document.getElementById('teamInputs');
                let html = '';
                for (let i = 1; i <= count; i++) {
                    html += `<input type="text" id="team${i}" placeholder="Tim ${i}" value="Tim ${i}">`;
                }
                container.innerHTML = html;
            });

            // Enter key to submit answer
            document.getElementById('answerInput').addEventListener('keypress', (e) => {
                if (e.key === 'Enter') submitAnswer();
            });
        }

        // ============ LOAD GAMES ============
        async function loadGames() {
            try {
                const res = await fetch(`${API_BASE}/api/game?page=1&perPage=50`);
                const json = await res.json();

                const gameList = document.getElementById('gameList');
                const games = json.data?.data || [];

                if (games.length === 0) {
                    gameList.innerHTML = `<div class="error-msg"><h3>Tidak ada game</h3><p>Belum ada game yang dibuat.</p></div>`;
                    return;
                }

                gameList.innerHTML = games.map(g => `
          <div class="game-option" onclick="selectGame('${g.id}', this)">
            <h4>${g.name}</h4>
            <p>${g.description || 'Tidak ada deskripsi'}</p>
          </div>
        `).join('');

            } catch (err) {
                document.getElementById('gameList').innerHTML = `
          <div class="error-msg">
            <h3>Gagal memuat game</h3>
            <p>Pastikan server backend sudah berjalan.</p>
          </div>
        `;
            }
        }

        function selectGame(id, el) {
            selectedGameId = id;
            document.querySelectorAll('.game-option').forEach(e => e.classList.remove('selected'));
            el.classList.add('selected');
            document.getElementById('btnStart').disabled = false;
        }

        // ============ START GAME ============
        async function startGame() {
            if (!selectedGameId) return alert('Pilih game terlebih dahulu!');

            // Setup teams
            const count = parseInt(document.getElementById('teamCount').value);
            teams = [];
            for (let i = 1; i <= count; i++) {
                const input = document.getElementById(`team${i}`);
                teams.push({ name: input?.value || `Tim ${i}`, score: 0 });
            }
            currentTeam = 0;

            // Show loading
            document.getElementById('setupScreen').style.display = 'none';
            document.getElementById('gameBoard').innerHTML = `<div class="loading"><div class="spinner"></div><p>Memuat permainan...</p></div>`;
            document.getElementById('gameBoard').style.display = 'block';

            try {
                const res = await fetch(`${API_BASE}/api/game/jeopardy/${selectedGameId}/play/public`);
                const json = await res.json();

                if (!json.data || !json.data.rounds) {
                    throw new Error('Data game tidak valid');
                }

                gameData = json.data;
                renderBoard();

            } catch (err) {
                document.getElementById('gameBoard').innerHTML = `
          <div class="error-msg">
            <h3>Gagal memuat game</h3>
            <p>${err.message}</p>
            <button class="btn btn-continue" onclick="resetGame()">Kembali</button>
          </div>
        `;
            }
        }

        // ============ RENDER BOARD ============
        function renderBoard() {
            // Show score bar
            document.getElementById('scoreBar').style.display = 'flex';
            document.getElementById('gameTitleHeader').textContent = gameData.name || '';
            renderScores();

            const round = gameData.rounds[0];
            if (!round?.categories) {
                document.getElementById('gameBoard').innerHTML = `<div class="error-msg"><h3>Game tidak memiliki kategori</h3></div>`;
                return;
            }

            const categories = round.categories;
            const numCats = categories.length;
            const maxClues = Math.max(...categories.map(c => c.clues?.length || 0));

            totalClues = categories.reduce((sum, c) => sum + (c.clues?.length || 0), 0);

            let html = '';

            // Category headers
            categories.forEach(c => {
                html += `<div class="category-header">${c.title}</div>`;
            });

            // Clue cells
            for (let row = 0; row < maxClues; row++) {
                categories.forEach(cat => {
                    const clue = cat.clues?.[row];
                    if (clue) {
                        const key = `${round.id}|${cat.id}|${clue.id}`;
                        html += `
              <div class="clue-cell" id="cell-${key}" onclick="openClue('${round.id}','${cat.id}','${clue.id}','${cat.title}',${clue.value})">
                $${clue.value}
              </div>
            `;
                    } else {
                        html += `<div class="clue-cell used"></div>`;
                    }
                });
            }

            const board = document.createElement('div');
            board.className = 'board';
            board.style.gridTemplateColumns = `repeat(${numCats}, 1fr)`;
            board.innerHTML = html;

            document.getElementById('gameBoard').innerHTML = '';
            document.getElementById('gameBoard').appendChild(board);
        }

        function renderScores() {
            document.getElementById('scoreBar').innerHTML = teams.map((t, i) => `
        <div class="team-score ${i === currentTeam ? 'active' : ''}">
          <div class="name">${t.name}</div>
          <div class="score">$${t.score.toLocaleString()}</div>
        </div>
      `).join('');
        }

        // ============ CLUE HANDLING ============
        function openClue(roundId, catId, clueId, catTitle, value) {
            const key = `${roundId}|${catId}|${clueId}`;
            if (usedClues.has(key)) return;

            // Find clue data
            const round = gameData.rounds.find(r => r.id === roundId);
            const cat = round?.categories?.find(c => c.id === catId);
            const clue = cat?.clues?.find(c => c.id === clueId);
            if (!clue) return;

            currentClue = { roundId, catId, clueId, key, value, question: clue.question, catTitle };

            // Show modal
            document.getElementById('clueValue').textContent = `$${value}`;
            document.getElementById('clueCategory').textContent = catTitle;
            document.getElementById('clueQuestion').textContent = clue.question;
            document.getElementById('answerInput').value = '';
            document.getElementById('answerSection').style.display = 'block';
            document.getElementById('resultBox').className = 'result-box';
            document.getElementById('clueModal').classList.add('active');
            document.getElementById('answerInput').focus();
        }

        async function submitAnswer() {
            const answer = document.getElementById('answerInput').value.trim();
            if (!answer) return;

            try {
                const res = await fetch(`${API_BASE}/api/game/jeopardy/${selectedGameId}/check`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        round_id: currentClue.roundId,
                        category_id: currentClue.catId,
                        clue_id: currentClue.clueId,
                        submitted_answer: answer
                    })
                });

                const json = await res.json();
                const result = json.data;

                if (result?.is_correct) {
                    showResult(true, result.correct_answer, result.points_earned);
                    teams[currentTeam].score += result.points_earned;
                } else {
                    showResult(false, result?.correct_answer || 'Tidak diketahui', 0);
                    teams[currentTeam].score -= currentClue.value;
                }
                renderScores();

            } catch (err) {
                showResult(false, 'Error memeriksa jawaban', 0);
            }
        }

        async function revealAnswer() {
            try {
                const res = await fetch(`${API_BASE}/api/game/jeopardy/${selectedGameId}/check`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        round_id: currentClue.roundId,
                        category_id: currentClue.catId,
                        clue_id: currentClue.clueId,
                        submitted_answer: ''
                    })
                });

                const json = await res.json();
                showResult(false, json.data?.correct_answer || 'Tidak diketahui', 0);
            } catch {
                showResult(false, 'Error', 0);
            }
        }

        function skipClue() {
            closeClue();
        }

        function showResult(correct, correctAnswer, points) {
            document.getElementById('answerSection').style.display = 'none';
            const box = document.getElementById('resultBox');
            const text = document.getElementById('resultText');
            const answerEl = document.getElementById('correctAnswerText');

            if (correct) {
                box.className = 'result-box correct';
                text.textContent = `âœ“ BENAR! +$${points}`;
            } else {
                box.className = 'result-box incorrect';
                text.textContent = `âœ— SALAH! -$${currentClue.value}`;
            }
            answerEl.textContent = `Jawaban: ${correctAnswer}`;
        }

        function closeClue() {
            usedClues.add(currentClue.key);
            const cell = document.getElementById(`cell-${currentClue.key}`);
            if (cell) {
                cell.classList.add('used');
                cell.textContent = '';
            }

            // Next team
            currentTeam = (currentTeam + 1) % teams.length;
            renderScores();

            document.getElementById('clueModal').classList.remove('active');
            currentClue = null;

            // Check game over
            if (usedClues.size >= totalClues) {
                endGame();
            }
        }

        // ============ END GAME ============
        function endGame() {
            document.getElementById('gameBoard').style.display = 'none';
            document.getElementById('scoreBar').style.display = 'none';

            const maxScore = Math.max(...teams.map(t => t.score));

            document.getElementById('finalScores').innerHTML = teams.map(t => `
        <div class="final-card ${t.score === maxScore ? 'winner' : ''}">
          <h3>${t.name}</h3>
          <div class="score">$${t.score.toLocaleString()}</div>
          ${t.score === maxScore ? '<div class="trophy">ðŸ† PEMENANG!</div>' : ''}
        </div>
      `).join('');

            document.getElementById('gameOver').style.display = 'block';
        }

        // ============ RESET GAME ============
        function resetGame() {
            gameData = null;
            teams = [];
            currentTeam = 0;
            currentClue = null;
            usedClues = new Set();
            totalClues = 0;
            selectedGameId = null;

            document.getElementById('setupScreen').style.display = 'flex';
            document.getElementById('gameBoard').style.display = 'none';
            document.getElementById('gameOver').style.display = 'none';
            document.getElementById('scoreBar').style.display = 'none';
            document.getElementById('gameTitleHeader').textContent = '';
            document.getElementById('btnStart').disabled = true;

            document.querySelectorAll('.game-option').forEach(e => e.classList.remove('selected'));
            loadGames();
        }
